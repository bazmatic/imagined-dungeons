# Currency Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Agents own gold (a number); items can carry a `priceTag`; players use `buy`/`sell`/`offer` to trade with NPCs, with an LLM consent check per trade.

**Architecture:** Add `Agent.gold` and `Item.priceTag` to the domain (and DB + UpsertInput + snapshot). Three new actions — buy, sell, offer — with parser cases, LLM-interpret integration, and admin UI controls. A small `tradeDecide` LLM call per trade returns `{accept, narration}`; an accepted trade atomically swaps gold and item ownership and clears the price tag.

**Tech Stack:** TypeScript strict mode, Drizzle ORM + better-sqlite3, Vitest, TanStack Start/React. Const-object discriminators only (no string literals in logic).

**Spec:** [docs/superpowers/specs/2026-05-13-currency-design.md](../specs/2026-05-13-currency-design.md)

import type { Agent, Item } from '@core/domain/entities';
import type { JsonSchema, LanguageModel } from './language-model';

export const TradeDirection = {
  Buy: 'buy',
  Sell: 'sell',
} as const;
export type TradeDirection = (typeof TradeDirection)[keyof typeof TradeDirection];

export interface TradeDecideRequest {
  readonly buyer: Agent;
  readonly seller: Agent;
  readonly item: Item;
  readonly price: number;
  readonly direction: TradeDirection;
}

export interface TradeDecision {
  readonly accept: boolean;
  readonly narration: string;
}

const SYSTEM_PROMPT = [
  'You decide whether a non-player character accepts a trade.',
  'Read the seller, the buyer, and the item description. Decide whether the seller would accept the offered price for that item, given their personality, mood, and goals.',
  'Respond with a JSON object containing two fields:',
  '  - accept: boolean — true to accept the trade, false to refuse.',
  '  - narration: string — a short in-character line from the seller (one or two sentences). On accept, it should sound like agreement; on refusal, like a polite or pointed decline.',
  'Never reveal mechanics in the narration. Speak in the seller\'s voice.',
].join('\n');

const SCHEMA: JsonSchema = {
  type: 'object',
  additionalProperties: false,
  required: ['accept', 'narration'],
  properties: {
    accept: { type: 'boolean' },
    narration: { type: 'string' },
  },
};

function buildUser(req: TradeDecideRequest): string {
  const directionLine =
    req.direction === TradeDirection.Buy
      ? `${req.buyer.label} wants to buy the ${req.item.label} from ${req.seller.label} for ${req.price} gold.`
      : `${req.buyer.label} (an NPC) is being asked to buy the ${req.item.label} from ${req.seller.label} (the player) for ${req.price} gold.`;
  return [
    directionLine,
    '',
    'Seller:',
    `  Label: ${req.seller.label}`,
    `  Short: ${req.seller.shortDescription}`,
    `  Long: ${req.seller.longDescription}`,
    `  Mood: ${req.seller.mood ?? '(none)'}`,
    `  Goal: ${req.seller.goal ?? '(none)'}`,

function coerce(parsed: unknown): TradeDecision | null {
  if (typeof parsed !== 'object' || parsed === null) return null;
  const r = parsed as Record<string, unknown>;
  if (typeof r.accept !== 'boolean') return null;
  if (typeof r.narration !== 'string') return null;
  return { accept: r.accept, narration: r.narration };
}

export async function tradeDecide(
  req: TradeDecideRequest,
  llm: LanguageModel,
): Promise<TradeDecision> {
  try {
    const response = await llm.complete({
      system: SYSTEM_PROMPT,
      user: buildUser(req),
      schema: SCHEMA,
      schemaName: 'trade_decision',
    });
    const decision = coerce(response.parsed);
    if (decision) return decision;
  } catch {
    // fall through to refuse
  }
  // Malformed payload or LLM error → refuse with a generic narration.
  return {
    accept: false,
    narration: `${req.seller.label} hesitates, then declines.`,
  };
}
